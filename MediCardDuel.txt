4. 题目加载器（按需加载）：
   ```javascript
   // src/modules/question-bank/question-loader.js

   class QuestionLoader {
     constructor() {
       this.loadedSubjects = new Set();
       this.questionCache = new Map();
       this.compressedCache = null;
     }

     // 加载指定学科
     async loadSubject(subjectId) {
       if (this.loadedSubjects.has(subjectId)) {
         return this.questionCache.get(subjectId);
       }

       // 从内嵌数据或异步获取
       const questions = await this.fetchSubjectData(subjectId);

       // 缓存
       this.questionCache.set(subjectId, questions);
       this.loadedSubjects.add(subjectId);

       return questions;
     }

     // 批量加载多个学科
     async loadSubjects(subjectIds) {
       const promises = subjectIds.map(id => this.loadSubject(id));
       return Promise.all(promises);
     }

     // 获取指定难度的题目
     getQuestionsByDifficulty(subjectId, difficulty) {
       const questions = this.questionCache.get(subjectId) || [];
       return questions.filter(q => q.difficulty === difficulty);
     }

     // 随机获取题目（用于出题）
     getRandomQuestion(subjectIds, difficulty, excludeIds = []) {
       const allQuestions = [];
       subjectIds.forEach(id => {
         const questions = this.questionCache.get(id) || [];
         allQuestions.push(...questions);
       });

       const filtered = allQuestions.filter(q => 
         q.difficulty === difficulty && 
         !excludeIds.includes(q.id)
       );

       return filtered[Math.floor(Math.random() * filtered.length)];
     }
   }
   ```

5. 单机/联机统一选择逻辑：
   ```javascript
   // 统一选择接口
   class SubjectSelector {
     constructor(mode) {  // 'single' | 'multiplayer'
       this.mode = mode;
       this.selectedSubjects = new Set();
     }

     // 选择学科（单机/联机通用）
     selectSubject(subjectId) {
       this.selectedSubjects.add(subjectId);
       this.updateUI();
       this.saveSelection();

       // 联机模式：同步给其他玩家
       if (this.mode === 'multiplayer' && this.isHost) {
         this.broadcastSelection();
       }
     }

     // 取消选择
     deselectSubject(subjectId) {
       this.selectedSubjects.delete(subjectId);
       this.updateUI();
       this.saveSelection();

       if (this.mode === 'multiplayer' && this.isHost) {
         this.broadcastSelection();
       }
     }

     // 获取已选题量统计
     getSelectionStats() {
       const stats = {
         totalQuestions: 0,
         byDifficulty: { common: 0, rare: 0, epic: 0, legendary: 0 },
         bySubject: {}
       };

       this.selectedSubjects.forEach(subjectId => {
         const questions = QuestionLoader.getSubject(subjectId);
         stats.totalQuestions += questions.length;
         questions.forEach(q => {
           stats.byDifficulty[q.difficulty]++;
         });
         stats.bySubject[subjectId] = questions.length;
       });

       return stats;
     }

     // 保存选择（单机：localStorage；联机：同步到Host）
     saveSelection() {
       localStorage.setItem('medicard_selected_subjects', 
         JSON.stringify([...this.selectedSubjects]));
     }

     // 恢复上次选择
     restoreSelection() {
       const saved = localStorage.getItem('medicard_selected_subjects');
       if (saved) {
         this.selectedSubjects = new Set(JSON.parse(saved));
       }
     }
   }
   ```
